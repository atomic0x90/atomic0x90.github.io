---

**설명**

[논리 연산자를 사용하는 법 알아보기][10]

---

**code**
```cpp
#include <iostream>

using namespace std;

int main()
{
        int a;

        cin>>a;

        if(a >= 90)
                cout<<"A"<<endl;
        else if(a < 90 && a >= 80)
                cout<<"B"<<endl;
        else if(a < 80 && a >= 70)
                cout<<"C"<<endl;
        else if(a < 70 && a >= 60)
                cout<<"D"<<endl;
        else
                cout<<"F"<<endl;
        return 0;
}
```
